Message.confirm({
    Msg: "确定删除吗？",
    okSkin: 'danger',
    iconImg: 'question',
    isModal: true
}).on(function (flag) {
    if (flag) {
        Message.alert({Msg: "删除", iconImg: 'info'});
    }else{
        Message.alert({Msg: "不删除", iconImg: 'info'});
    }
});
